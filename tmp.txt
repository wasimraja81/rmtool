# Dynamically find the absolute directory where this script and SIF file live
image_dir="$(dirname "$(readlink -f "$0")")"
image_name="misc_rmtool-1.0.sif"

# The command matches the symlink clicked (e.g., rm_synthesis_OMP1_GPU0)
cmd="$(basename "$0")"
args="$@"

# Automatically add NVIDIA GPU drivers only when the GPU binary is requested
nv_flag=""
if [[ "$cmd" == *"GPU1"* ]]; then
    nv_flag="--nv"
fi

# Execute with a clean environment (-e) and use absolute internal target paths
singularity exec \
    -e \
    $nv_flag \
    -B ~/.Xauthority \
    "$image_dir/$image_name" "/app/rmtool/bin/$cmd" $args
